        // Aggiorna la tabella pivot

        function updatePivotTable() {

            const rowGroup = document.getElementById('rowGroup').value;

            const colGroup = document.getElementById('colGroup').value;

            const valueType = document.getElementById('valueType').value;

            

            // Crea la matrice pivot

            const pivotData = {};

            const rowValues = [...new Set(filteredData.map(d => d[rowGroup]))].sort();

            const colValues = [...new Set(filteredData.map(d => d[colGroup]))].sort();

            

            filteredData.forEach(d => {

                const row = d[rowGroup];

                const col = d[colGroup];

                

                if (!pivotData[row]) pivotData[row] = {};

                if (!pivotData[row][col]) pivotData[row][col] = [];

                

                pivotData[row][col].push(d);

            });

            

            // Genera la tabella HTML

            let html = '<table><thead><tr><th>' + rowGroup.charAt(0).toUpperCase() + rowGroup.slice(1) + '</th>';

            colValues.forEach(col => {

                html += `<th>${col}</th>`;

            });

            html += '<th>Totale</th></tr></thead><tbody>';

            

            rowValues.forEach(row => {

                html += `<tr><td style="font-weight:bold;background:#f8f9fa;">${row}</td>`;

                let rowTotal = 0;

                

                colValues.forEach(col => {

                    const cellData = pivotData[row] && pivotData[row][col] ? pivotData[row][col] : [];

                    const count = cellData.length;

                    rowTotal += count;

                    

                    if (valueType === 'count') {

                        const cellClass = count > 0 ? 'number' : '';

                        const editableClass = editMode && count > 0 ? 'editable' : '';

                        html += `<td class="${cellClass} ${editableClass}" style="cursor:pointer;" onclick="showDetail('${row}', '${col}')">${count || ''}</td>`;

                    } else {

                        const truncatedList = cellData.slice(0, 3).map(d => d.conoscenza.substring(0, 40) + '...').join('<br>');

                        const moreText = cellData.length > 3 ? `<br><i>...e altri ${cellData.length - 3}</i>` : '';

                        const editableClass = editMode && cellData.length > 0 ? 'editable' : '';

                        html += `<td class="${editableClass}" style="font-size:0.8em;cursor:pointer;" onclick="showDetail('${row}', '${col}')">${truncatedList}${moreText}</td>`;

                    }

                });

                

                html += `<td class="number" style="font-weight:bold;background:#ecf0f1;">${rowTotal}</td></tr>`;

            });

            

            // Riga totali

            html += '<tr style="font-weight:bold;background:#34495e;color:white;"><td>Totale</td>';

            let grandTotal = 0;

            colValues.forEach(col => {

                const colTotal = rowValues.reduce((sum, row) => {

                    const cellData = pivotData[row] && pivotData[row][col] ? pivotData[row][col] : [];

                    return sum + cellData.length;

                }, 0);

                grandTotal += colTotal;

                html += `<td class="number">${colTotal}</td>`;

            });

            html += `<td class="number">${grandTotal}</td></tr>`;

            

            html += '</tbody></table>';

            document.getElementById('pivotTable').innerHTML = html;

        }

        function editCell(cell, dataIndex, field) {

            if (!editMode) return;

            

            const currentValue = cell.textContent;

            const input = document.createElement('input');

            input.type = 'text';

            input.value = currentValue;

            input.className = 'edit-input';

            

            input.onblur = function() {

                const newValue = this.value.trim();

                if (newValue && newValue !== currentValue) {

                    rawData[dataIndex][field] = newValue;

                    cell.textContent = newValue;

                    filterData();

                    showMessage('✅ Modifiche salvate!', 'success');

                } else {

                    cell.textContent = currentValue;

                }

            };

            

            input.onkeypress = function(e) {

                if (e.key === 'Enter') {

                    this.blur();

                }

            };

            

            cell.innerHTML = '';

            cell.appendChild(input);

            input.focus();

            cell.classList.add('editing');

        }
